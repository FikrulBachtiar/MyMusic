  late YoutubePlayerController _playerController;
  final PanelController _panelController = PanelController();

  @override
  void initState() {
    _pagingController.addPageRequestListener((page) {
      fetchPage(page);
    });
    servedProfile.init();
    super.initState();
  }

  @override
  void dispose() {
    _pagingController.dispose();
    _playerController.dispose();
    Get.delete<BerandaObs>();
    super.dispose();
  }

  fetchPage(int page) async {
    BerandaService()
        .getVideoList(
      maxResults: served.sizeList.value,
      pageToken: served.pageToken.value,
      videoCategoryId: served.videoCategoryId.value.toString(),
    )
        .then(
      (res) {
        List<Item>? data = res.items ?? [];
        var document = parse(data[0].player?.embedHtml);
        var urlVideo = document.querySelectorAll('iframe')[0].attributes["src"];
        if (res.nextPageToken == null) {
          _pagingController.appendLastPage(data);
        } else {
          served.pageToken.value = res.nextPageToken.toString();
          _pagingController.appendPage(data, page);
        }
        if (data.length >= 0) {
          _playerController = YoutubePlayerController(
            initialVideoId: data[0].id.toString(),
            flags: const YoutubePlayerFlags(),
          );
        }
      },
    );
  }

  // youtubeListener() {

  @override
  void deactivate() {
    _playerController.pause();
    super.deactivate();
  }

  playVideos(Item items) {
    _playerController.load(items.id.toString());
    served.isOpen.value = true;
    served.minHeightPanel.value = 70;
    _panelController.show();
    _panelController.open();
  }

  Widget playVideoViews(ScrollController scrollController) {
    return Obx(
      () {
        return served.isOpen.isFalse
            ? Container()
            : Container(
                color: const Color.fromRGBO(33, 33, 33, 1),
                child: Column(
                  children: [
                    YoutubePlayerBuilder(
                      onExitFullScreen: () {
                        SystemChrome.setPreferredOrientations(
                            DeviceOrientation.values);
                      },
                      player: YoutubePlayer(
                        controller: _playerController,
                        onReady: () {
                          print("HELLE");
                        },
                        showVideoProgressIndicator: true,
                        progressColors: const ProgressBarColors(
                          playedColor: Colors.red,
                          handleColor: Colors.red,
                        ),
                      ),
                      builder: (_, widgets) {
                        return widgets;
                      },
                    ),
                  ],
                ),
              );
      },
    );
  }
